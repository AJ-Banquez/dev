"SELECT 
  SYSADM.PS_V_PERSONAL_NAME.EMPLID  AS   Cedula, --""Cedula Empleado"",
  SYSADM.PS_V_PERSONAL_NAME.NAME As Nombre_Completo,
  SYSADM.PS_PERSONAL_DATA.SEX,
  SYSADM.PS_PERSONAL_DATA.BIRTHDATE Fecha_Nacimiento,
  SYSADM.PS_PERSONAL_DATA.BIRTHPLACE Municipio_Nac,
  SYSADM.PS_PERSONAL_DATA.BIRTHSTATE Departamento_Nac,
  SYSADM.PS_PERSONAL_DATA.CITY Ciudad_Recidencia,
  SYSADM.PS_PERSONAL_DATA.STATE Departameto_Recidencia,
  SYSADM.PS_V_PUESTO.DES_TITL_PUEST As Puesto,
  decode(SYSADM.PS_V_EMPLM_SRCH_Q.PAYGROUP ,'0','PTC','1','PTC','2','PTC','MPT')As Rol,
  SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_DIVSN As Codigo_Divicion,
  DIVISIONES.DESCR As Divicion,
  SYSADM.PS_V_EMPLM_SRCH_Q.DESCR_PUESTO As Descripcion_del_Cargo,
  SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_DEPRT As Codigo_Departamento,
  SYSADM.PS_V_EMPLM_SRCH_Q.COD_NOMN,
  DEPARTAMENTOS.DESCR As DEPARTAMENTO,
  SYSADM_PS_PERSONAL_DATA_SUPER.NAME As Nombre_Supervisor,
  SYSADM_PS_PERSONAL_DATA_SUPER.PHONE As Telefono_Supervisor,
  CUADRILLAS.COD_SUPRV As Linea,
  SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_SUPRN AS Codigo_SUPERINTENDENCIA,
  CUADRILLAS.DESCR AS CUADRILLA,
  GERENCIAS.DESCR As GERENCIAS,
  SUPERINTENDENCIAS.DESCR as SUPERINTENDENCIA,
  VICEPRESIDENCIAS.DESCR VICEPRESIDENCIA,
  SYSADM.PS_B_EMPLEADO.fec_ingrs,
  SYSADM.PS_B_EMPLEADO.fec_termn_contr


FROM  
           SYSADM.PS_V_PERSONAL_NAME               
INNER JOIN SYSADM.PS_V_EMPLM_SRCH_Q                ON SYSADM.PS_V_EMPLM_SRCH_Q.EMPLID = SYSADM.PS_V_PERSONAL_NAME.EMPLID
INNER JOIN SYSADM.PS_PERSONAL_DATA                ON SYSADM.PS_PERSONAL_DATA.EMPLID = SYSADM.PS_V_PERSONAL_NAME.EMPLID
INNER JOIN SYSADM.PS_V_DEPT_TBL  GERENCIAS         ON SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_GERNC = GERENCIAS.DEPTID 
INNER JOIN SYSADM.PS_V_DEPT_TBL  DEPARTAMENTOS     ON SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_DEPRT = DEPARTAMENTOS.DEPTID
INNER JOIN SYSADM.PS_V_DEPT_TBL  VICEPRESIDENCIAS  ON SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_VICPR = VICEPRESIDENCIAS.DEPTID
INNER JOIN SYSADM.PS_V_DEPT_TBL  SUPERINTENDENCIAS ON SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_SUPRN = SUPERINTENDENCIAS.DEPTID
INNER JOIN SYSADM.PS_V_DEPT_TBL  DIVISIONES        ON SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_DIVSN = DIVISIONES.DEPTID
INNER JOIN SYSADM.PS_V_DEPT_TBL  CUADRILLAS        ON SYSADM.PS_V_EMPLM_SRCH_Q.COD_DEPND_CUADR = CUADRILLAS.DEPTID 
INNER JOIN SYSADM.PS_B_EMPLEADO                    ON SYSADM.PS_V_PERSONAL_NAME.EMPLID = SYSADM.PS_B_EMPLEADO.EMPLID
INNER JOIN SYSADM.PS_V_PUESTO                      ON SYSADM.PS_V_PUESTO.COD_PUEST = SYSADM.PS_V_EMPLM_SRCH_Q.COD_PUEST
LEFT  JOIN SYSADM.PS_V_SUPERVISOR                  ON SYSADM.PS_V_EMPLM_SRCH_Q.EMPLID = SYSADM.PS_V_SUPERVISOR.EMPLID
LEFT  JOIN SYSADM.PS_PERSONAL_DATA  SYSADM_PS_PERSONAL_DATA_SUPER ON SYSADM.PS_V_SUPERVISOR.EMPLID_SUPRV = SYSADM_PS_PERSONAL_DATA_SUPER.EMPLID

WHERE
    SYSADM.PS_V_EMPLM_SRCH_Q.COD_NOMN  IN  ( '1'  )
    "
